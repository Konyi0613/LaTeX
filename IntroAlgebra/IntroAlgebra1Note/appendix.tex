\begin{theorem}
If \(H, K < G\), then 
\[
    \left\vert HK \right\vert = \frac{\vert H \vert \cdot \vert K \vert }{\left\vert H \cap K \right\vert }.
\]   
\end{theorem}
\begin{theorem}
    If \(A, B \triangleleft G\), then \(AB \triangleleft G\).  
\end{theorem}
\begin{proof}
    \vphantom{text}
    \begin{itemize}
        \item \(AB < G\):
        \[
            a_1 b_1 a_2 b_2 = a_1 \left( b_1 a_2 b_1^{-1} \right) b_1 b_2 \in AB, \quad \left( ab \right)^{-1} = b^{-1} a^{-1} = \left( b^{-1} a^{-1} b \right)  b^{-1} \in AB.
        \]
        \item \(AB \triangleleft G\): 
        \[
            gABg^{-1} = \left( gAg^{-1} \right)\left( gBg^{-1} \right) = AB.  
        \]
    \end{itemize}
\end{proof}

\begin{theorem}
    If \(A, B \triangleleft G\) and \(A \cap B = \left\{ e \right\} \), then \(AB \simeq A \times B\).   
\end{theorem}
\begin{proof}
Define \(\varphi :A \times B \to AB\) by \(\varphi \left( a, b \right) = ab \), and this is the isomorphism.  
\end{proof}
