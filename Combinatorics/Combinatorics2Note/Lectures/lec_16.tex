\begin{theorem}[Integrality theorem]
    If \(c\) is integer-valued, then there is a max flow that is integer-valued. 
\end{theorem}
\begin{proof}
    Ford-Fulkerson algorithm: all tolerances are integer-valued. 
    \begin{remark}
        Max flow do not needs to be integer-valued.
    \end{remark}
\end{proof}

\begin{theorem}[Unit flow decomposition]
    Given \((\vv{D}, s, t, c)\) and any feasible integer-valued flow \(f\), we can decompose \(f\) into unit flow along \(s, t\)-paths (plus closed loops).     
\end{theorem}
\begin{proof}
    Keep the edges with positive flow.
\end{proof}

\paragraph{Baranyai's Theorem}
Suppose the league has 22 teams. Every weekend some games are played: Every team plays at most once in a weekend. Besides, every pair of temas should play exactly once in the season. 
\begin{question}
    How long must the season be?
\end{question}
\textbf{Lower bound:} At least 21 weekends since any team must play all 21 others.

Now we can convert the tournament into a complete graph \(K_{22}\), with vertices representing teams and edges representing matches, and we can color edges according to the schedule, e.g. \textcolor{red}{Week 1}, \textcolor{green}{Week 2}, \textcolor{blue}{Week 3}, \textcolor{yellow}{Week 4}.

Note that, in order for every team to play every weekend, the edges of each color should be a perfect matching; on the other hand, we want to avoid repetitive matches between the same pair teams, and so an edge can only have one color. The question thus becomes as follows.
\begin{question}
    Can we decompose \(E(K_n)\) into perfect matchings? 
\end{question}

There is a clever construction to decompose $E(K_n)$ into perfect matchings by arranging the vertices into a regular $(n-1)$-gon plus a central vertex. Then, we can match the central vertex with a vertex and pair up the rest, and then rotate $(n-1)$ times. (See next page)

\begin{definition}
    A \(k\)-graph (\(k\)-uniform hypergraph) is a pair \((V, E)\) where \(E \subseteq \binom{V}{k}\), i.e. every edge contains \(k\) vertices. The \(k\)-graph is complete if \(E = \binom{V}{k}\). A matching is a pairwise-disjoint set of edges and is perfect if every vertex is covered.       
\end{definition}

\begin{theorem}[Baranyai, 1973]
    The edges of a complete \(k\)-graph on \(n\) vertices, where \(k \mid n\), can be partitioned into perfect matchings.   
\end{theorem}

\begin{question}
    How many edges are there?
\end{question}
\begin{answer}
    \(\binom{n}{k}\). 
\end{answer}

\begin{question}
    How many edges in a perfect matching?
\end{question}
\begin{answer}
    \(\frac{n}{k}\). 
\end{answer}

\begin{question}
    How many perfect matchings do we need?
\end{question}
\begin{answer}
    \(\frac{\binom{n}{k}}{\left( \frac{n}{k} \right) } = \binom{n - 1}{k - 1}\). 
\end{answer}

\begin{question}
    How many perfect matchings are there?
\end{question}
\begin{answer}
    \(\binom{n}{k} \binom{n - k}{k} \dots \binom{k}{k} \cdot \frac{1}{\left( \frac{n}{k} \right)! }\). 
\end{answer}

\paragraph{Special cases:}
\begin{itemize}
    \item \(k = 3\): Pelte sohn, 1936. 
    \item \(k = 4\): Bermond. 
    \item all \(k\): Baranyai, 1973.   
\end{itemize}

\begin{proof}[proof of Baranyai's theorem]
    We first give a definition:
    \begin{definition}
        A \(m\)-partition of \([n]\), \(\mathcal{A}\), is a partition of \([n]\) into \(m\) pairwise-disjoint parts (multiple parts could be empty), e.g.
        \[
            \underbrace{\varnothing , \varnothing , \dots , \varnothing }_{m - 1}, [n]
        \]   
        is an \(m\)-partition. 
    \end{definition}

    \begin{eg}
        A perfect matching in the complete \(k\)-graph is an \(m\)-partition of \(V = [n]\) where \(m = \frac{n}{k}\).    
    \end{eg}

    We want \(M = \binom{n - 1}{k - 1}\) perfect matchings. 
    \begin{claim}
        For any \(0 \le \ell \le n\), there is a collection \(\mathcal{A}_1, \mathcal{A}_2, \dots , \mathcal{A}_M\) of \(m\)-partitions of \([\ell]\) s.t. \(m = \frac{n}{k}\) and for any \(S \in 2^{[\ell]}\), \(S\) appears \(\binom{n - \ell}{k - \vert S \vert }\) times in \(\mathcal{A}_1, \mathcal{A}_2, \dots , \mathcal{A}_M\).       
    \end{claim}
    \textbf{proof:} We do induction on \(\ell\). 
        \begin{itemize}
            \item Base case: \(\ell = 0\), then let
            \[
                \mathcal{A}_i = \underbrace{\varnothing , \varnothing , \dots , \varnothing }_{m \text{ times}},
            \]
            we know \(\varnothing \) appears \(mM = \frac{n}{k} \binom{n - 1}{k - 1} = \binom{n}{k}\) times, so the claim is true for \(\ell = 0\). 
            \item Induction step: \(\ell \to \ell + 1\). Suppose we have a network \(\vv{D}\), where the source vertex \(s \to \mathcal{A}_i\) for \(i = 1, 2, \dots , M\), and \(\mathcal{A}_i \to S\) for \(S \in 2^{[\ell]}\) if \(S\) is a part of \(\mathcal{A}_i\), and \(S \to t\) for all \(S \in 2^{[\ell]}\) where \(t\) is the sink vertex. Now let 
            \[
                c(\vv{e}) = \begin{dcases}
                    1, &\text{ if } \vv{e} = s \to \mathcal{A}_i ;\\
                    \infty, &\text{ if } \vv{e} = \mathcal{A}_i \to S ;\\
                    \binom{n - \ell - 1}{k - \vert S \vert - 1}, &\text{ if } \vv{e} = S \to t.
                \end{dcases}
            \]
            Now we have \(M\) \(m\)-partitions of \([\ell]\), with the sets appearing with the right multiplicity. Now we need to choose to put \(\ell + 1\) in each \(m\)-partition, maintaining correct multiplicities.

            Network represents the assignment of \(\ell + 1\) to a part in each partition. Need to ensure that the set multiplicities are correct. 
            
            \textbf{Dream:} Find an integer flow with edges \((s, \mathcal{A}_i)\) and \((S, t)\) used to capacity. Then each \(\mathcal{A}_i\) has a part \(S_i \in \mathcal{A}_i\) s.t. \(f((\mathcal{A}_i, S_i)) = 1\). Hence, we can form new \(m\)-partition of \([\ell + 1]\) by taking \(\mathcal{A}_i\) and replacing \(S_i\) with \(S_i \cup \left\{ \ell + 1 \right\} \). Now we count the multiplicities of sets \(S \in 2^{[\ell + 1]}\). If \(\ell + 1 \in S\), then let \(S^{\prime} = S - \left\{ \ell + 1 \right\} \), we get \(S\) by adding \(\ell + 1\) to \(S^{\prime} \) in an \(m\)-partition. This happens 
            \[
                f((S^{\prime}, t)) = c((S^{\prime} , t)) = \binom{n - \ell - 1}{k - \vert S^{\prime}  \vert - 1} = \binom{n - (\ell + 1)}{k - \vert S \vert } \text{ times}. 
            \]                  
            Now if \(\ell + 1 \notin S\), we get \(S\) by having it in our \(m\)-partition before, and not adding \(\ell + 1\) to it. Hence, its multiplicity is 
            \begin{align*}
               \binom{n - \ell}{k - \vert S \vert } - \underbrace{f((S, t))}_{\substack{\text{\# of times} \\ \text{we add }  \ell + 1 \text{ to } S}} &= \binom{n - \ell}{k - \vert S \vert } - c((S, t)) \\
               &= \binom{n - \ell}{k - \vert S \vert } - \binom{n - \ell - 1}{k - \vert S \vert - 1} = \binom{n - \ell - 1}{k - \vert S \vert }.
            \end{align*}   

            Now we want to show that such a flow of value \(M\) exists in the original network.
            
            \textbf{Sanity checks:} Note that the value of this flow leaving \(s\) is \(M = \binom{n - 1}{k - 1}\), and the value of flow entering \(t\) is 
            \[
                \sum_{S} c((S, t)) = \sum_{S \subseteq [\ell]} \binom{n - \ell - 1}{k - \vert S \vert - 1} = \sum_{i=0}^{\ell } \binom{\ell}{i} \binom{n - \ell - 1}{k - i - 1} = \binom{n - 1}{k - 1} = M,    
            \]   
            which makes sense. Hence, we need a feasible integer flow of value \(\binom{n - 1}{k - 1}\).
            
            By Ford-Fulkerson, it is enough to show that \(\mathrm{cap}(S, \overline{S}) \ge \binom{n - 1}{k - 1} \) for any source sink cut \([S, \overline{S}]\), then max flow is equal to the min cut \(\ge \binom{n - 1}{k - 1}\), and thus the max flow is of value \(M\), and by integrality theorem, there exist an integer-valued flow of value \(\binom{n - 1}{k - 1}\), and we're done. However, this is hard to prove since the edges between \(\mathcal{A}_i\) and \(S\) are of capacity \(\infty\) by our setting, so it is hard to analyze the capacity of a source sink cut.   

            The actual argument is to find a fractional feasible flow \(f^*\) with \(\mathrm{val}(f^*) = \binom{n - 1}{k - 1} \), then max flow \(\ge \binom{n - 1}{k - 1}\), and by integrality theorem, there is an integer flow of value \(\binom{n - 1}{k - 1}\). Note that \(m\)-partition will grow into \(m = \frac{n}{k}\) sets of size \(k\). We add \(\ell + 1\) to a uniformly random spot, then 
            \[
                \mathbb{P} (\text{we add it to } S) = \frac{k - \vert S \vert }{n - \ell}.
            \]             
            Set 
            \[
                f^*((\mathcal{A}_i, S)) = \begin{dcases}
                    \frac{k - \vert S \vert }{n - \ell}, &\text{ if } \forall S \in \mathcal{A}_i, S \neq \varnothing ;\\
                    \frac{\lambda k}{n - \ell }, &\text{ if }\varnothing \in \mathcal{A}_i \text{ with multiplicity } \lambda.
                \end{dcases}
            \]
            Note that this setting is enough since there is an edge between \(\mathcal{A}_i\) and \(S\) iff \(S \in \mathcal{A}_i\). Now we analyze the flow value going out each vertex \(\mathcal{A}_i\) and \(S\).     
            Flow out of \(\mathcal{A}_i\):
            \begin{align*}
                \sum_{S \in \mathcal{A}_i} f^*((\mathcal{A}_i, S)) &= \sum_{\varnothing \neq S \in \mathcal{A}_i} \frac{k - \vert S \vert }{n - \ell} + \frac{\lambda k}{n - \ell} = \frac{\sum_{S \in \mathcal{A}_i} (k - \vert S \vert )}{n - \ell} \\
                &= \frac{\sum_{S \in \mathcal{A}_i} k - \sum_{S \in \mathcal{A}_i} \vert S \vert  }{n - \ell} = \frac{n - \ell}{n - \ell } = 1.   
            \end{align*} 
            Flow into \(S\):
            \[
                \sum_{i: S \in \mathcal{A}_i} f^*((A_i, S)) = \sum_{i: S \in \mathcal{A}_i} \frac{k - \vert S \vert }{n - \ell} = \frac{k - \vert S \vert }{n - \ell } \binom{n - \ell}{k - \vert S \vert } = \binom{n - \ell - 1}{k - \vert S \vert - 1}.  
            \]

    \begin{observation}
        \(\ell = n\) gives Baranyai's theorem. This is because 
        \[
            \binom{\ell - \ell}{k - \vert S \vert } = \begin{dcases}
                0, &\text{ if } k - \vert S \vert \neq 0  ;\\
                1, &\text{ if } k - \vert S \vert = 0.
            \end{dcases},
        \]
        i.e. \(S \in 2^{[\ell]}\) appears in \(\mathcal{A}_1, \dots , \mathcal{A}_M\) for exactly one time iff \(\vert S \vert = k \), otherwise \(S\) appears 0 time.  
    \end{observation}
\end{proof}
